cmake_minimum_required(VERSION 3.10)
project(LocalVideoServer C)

set(CMAKE_C_STANDARD 99)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Windows specific settings
if(WIN32)
    add_definitions(-D_WIN32_WINNT=0x0601)  # Windows 7+
    add_definitions(-DWIN32_LEAN_AND_MEAN)
    
    # Link Windows libraries
    set(WINDOWS_LIBS ws2_32 winmm pthread)
    
    # Disable warnings that are too strict for cross-platform code
    add_compile_options(
        -Wall
        -Wextra
        -Wno-unused-parameter
        -Wno-sign-compare
    )
else()
    add_compile_options(
        -Wall
        -Wextra
        -Werror
        -Wshadow
        -Wpointer-arith
        -Wcast-qual
        -Wstrict-prototypes
        -Wmissing-prototypes
        -Wmissing-declarations
        -Wno-format-truncation
    )
endif()

# Platform abstraction source
set(PLATFORM_SOURCES
    src/shared/platform/platform.c
)

# Core sources (excluding video_scanner for Windows)
set(SOURCES
    src/main.c
    src/shared/module/module.c
    src/shared/notifier/notifier.c
    src/shared/log/log.c
    src/shared/thread/thread.c
    src/shared/json/json.c
    src/modules/config/config.c
    src/modules/db_manager/db_manager.c
    src/modules/http_server/http_server.c
    src/modules/http_server/http_request.c
    src/modules/http_server/http_response.c
    src/modules/api_handler/api_handler.c
    ${PLATFORM_SOURCES}
)

# Include directories
include_directories(
    src/include
    src/shared/module
    src/shared/list
    src/shared/notifier
    src/shared/log
    src/shared/thread
    src/shared/json
    src/shared/platform
    src/modules/config
    src/modules/db_manager
    src/modules/http_server
    src/modules/api_handler
)

# Find SQLite3
find_package(SQLite3 REQUIRED)

# Main executable
add_executable(local_video ${SOURCES})

# Platform-specific linker options
if(NOT WIN32)
    target_link_options(local_video PRIVATE -T ${CMAKE_SOURCE_DIR}/linker.ld)
endif()

# Link libraries
target_link_libraries(local_video
    ${WINDOWS_LIBS}
    SQLite::SQLite3
)

# Windows executable settings
if(WIN32)
    set_target_properties(local_video PROPERTIES
        WIN32_EXECUTABLE FALSE
    )
endif()

set_target_properties(local_video PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)
